        // YouTube video loader
        function loadYouTubeVideo() {
            const urlInput = document.getElementById('youtubeUrl');
            const url = urlInput.value.trim();

            if (!url) {
                alert('Por favor, ingresa un enlace de YouTube válido');
                return;
            }

            // Extract video ID from YouTube URL
            let videoId = '';
            const regExp = /^.*(youtu.be\/|v\/|u\/\w\/|embed\/|watch\?v=|&v=)([^#&?]*).*/;
            const match = url.match(regExp);

            if (match && match[2].length === 11) {
                videoId = match[2];
            } else {
                alert('Por favor, ingresa un enlace de YouTube válido');
                return;
            }

            // Hide placeholder and show video
            document.getElementById('videoPlaceholder').classList.add('hidden');
            document.getElementById('youtubeContainer').classList.remove('hidden');
            document.getElementById('videoControls').classList.add('hidden');

            // Set video source
            const iframe = document.getElementById('youtubeVideo');
            iframe.src = `https://www.youtube.com/embed/${videoId}?autoplay=0&rel=0&modestbranding=1`;

            // Add reset button
            const videoSection = document.getElementById('video');
            const resetButton = document.createElement('div');
            resetButton.className = 'text-center mt-4';
            resetButton.innerHTML = `
                <button onclick="resetVideo()" class="text-green-600 hover:text-green-700 font-medium">
                    Cambiar video
                </button>
            `;
            videoSection.querySelector('.container').appendChild(resetButton);
        }

        // Reset video function
        function resetVideo() {
            document.getElementById('videoPlaceholder').classList.remove('hidden');
            document.getElementById('youtubeContainer').classList.add('hidden');
            document.getElementById('videoControls').classList.remove('hidden');
            document.getElementById('youtubeUrl').value = '';

            // Remove reset button
            const resetButton = document.querySelector('#video button[onclick="resetVideo()"]');
            if (resetButton) {
                resetButton.parentElement.remove();
            }
        }
